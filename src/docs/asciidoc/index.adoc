[plantuml,diagramaContexto,svg]
....

[*] --> State0
State0 --> State1 : Start
State1 --> State1 : Movement
State1 --> State1 : [undoable]Undo
State1 --> State1 : [redoable]Redo
State1 --> State3 : Exit
State3 --> State4 : Save
State1 --> State3 : [isTicTacToe]
State0 --> State4 : [isTicTacToe]
State2 --> State0 : [isResumed]
State4 --> State2 : Resume
State2 --> [*] : [else]
